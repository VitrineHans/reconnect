    -- New columns on profiles
    alter table public.profiles
      add column if not exists push_token text,
      add column if not exists onboarding_answers jsonb;

    -- profiles INSERT policy (trigger uses security definer but add explicit policy as fallback)

      on public.profiles for insert
      with check (auth.uid() = id);

    -- friend_invites: INSERT policy (from_user must be caller)
    create policy "Users can send invites"
      on public.friend_invites for insert
      with check (auth.uid() = from_user);

    -- friend_invites: UPDATE policy (to_user can accept/decline)
    create policy "Users can respond to invites"
      on public.friend_invites for update
      using (auth.uid() = to_user);

    -- friend_invites: DELETE policy (sender can withdraw)
    create policy "Users can cancel their own invites"
      on public.friend_invites for delete
      using (auth.uid() = from_user);

    -- friendships: INSERT policy (trigger uses security definer; client fallback)
    create policy "Friends can create friendships"
      on public.friendships for insert
      with check (auth.uid() = user_a or auth.uid() = user_b);

    -- Trigger: auto-create profile row when auth.users row is inserted
    create or replace function public.handle_new_user()
    returns trigger language plpgsql security definer set search_path = public as $$
    begin
      insert into public.profiles (id, username, display_name)
      values (new.id, '', '')
      on conflict (id) do nothing;
      return new;
    end;
    $$;

    drop trigger if exists on_auth_user_created on auth.users;
    create trigger on_auth_user_created
      after insert on auth.users
      for each row execute procedure public.handle_new_user();

    -- Trigger: create friendship when invite accepted (uses least/greatest for user_a < user_b)
    create or replace function public.handle_invite_accepted()
    returns trigger language plpgsql security definer as $$
    begin
      if new.status = 'accepted' and old.status = 'pending' then
        insert into public.friendships (user_a, user_b, streak_count)
        values (
          least(new.from_user, new.to_user),
          greatest(new.from_user, new.to_user),
          0
        )
        on conflict (user_a, user_b) do nothing;
      end if;
      return new;
    end;
    $$;

    drop trigger if exists on_invite_accepted on public.friend_invites;
    create trigger on_invite_accepted
      after update on public.friend_invites
      for each row execute procedure public.handle_invite_accepted();

    insert into storage.buckets (id, name, public)
    values ('avatars', 'avatars', true)
    on conflict (id) do nothing;

    create policy "Authenticated users can upload avatars"
      on storage.objects for insert
      with check (bucket_id = 'avatars' and auth.uid() is not null);

    create policy "Anyone can view avatars"
      on storage.objects for select

    create policy "Users can update own avatar"

    1. Create mobile/hooks/useProfile.ts:
    ```typescript
    import { useEffect, useState } from 'react';
    import { Session } from '@supabase/supabase-js';
    import { supabase } from '../lib/supabase';

    export interface Profile {
      id: string;
      username: string;
      display_name: string | null;
      avatar_url: string | null;
      push_token: string | null;
      onboarding_answers: Record<string, unknown> | null;
    }

    export function useProfile(session: Session | null): { profile: Profile | null; profileLoading: boolean; refetch: () => void } {
      const [profile, setProfile] = useState<Profile | null>(null);
      const [profileLoading, setProfileLoading] = useState(true);

      const fetchProfile = async () => {
        if (!session?.user?.id) {
          setProfile(null);
          setProfileLoading(false);
          return;
        }
        setProfileLoading(true);
        const { data, error } = await supabase
          .from('profiles')
          .select('id, username, display_name, avatar_url, push_token, onboarding_answers')
          .eq('id', session.user.id)
          .single();
        if (!error) setProfile(data);
        setProfileLoading(false);
      };

      useEffect(() => {
        fetchProfile();
      }, [session?.user?.id]);

      return { profile, profileLoading, refetch: fetchProfile };
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→Supabase Auth | Email input and OTP code cross from untrusted mobile client to Supabase Auth service |
| client→profiles table | Profile update (display_name, avatar_url, onboarding_answers) crosses from client to RLS-gated table |
| client→storage | Avatar image binary uploaded from client to Supabase Storage bucket |
| auth guard→router | Route segment read from Expo Router then used to trigger navigation — no external input but must handle loading race |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Spoofing | supabase.auth.signInWithOtp | accept | Supabase handles OTP generation and delivery; codes are single-use, 10-min expiry by default. User enumeration mitigated: Supabase returns identical response whether email exists or not. |
| T-01-02 | Spoofing | supabase.auth.verifyOtp | mitigate | OTP maxLength=6, keyboardType=number-pad prevents non-numeric input on client. Brute-force protection handled by Supabase Auth (rate limits per email). No client-side secret stored — only session tokens returned by verifyOtp. |
| T-01-03 | Tampering | profiles UPDATE (display_name, username) | mitigate | RLS policy `using (auth.uid() = id)` on UPDATE enforces server-side. Client-side validate username with /^[a-zA-Z0-9_]{3,20}$/ (V5 input validation). display_name length check 1-50 client-side + DB text type has no length limit (acceptable — display only). |
| T-01-04 | Information Disclosure | profiles SELECT using (true) | accept | Intentional — required for username search (FRIEND-01). Only username, display_name, avatar_url exposed — no email, phone, or internal IDs beyond profile UUID. Email remains in auth.users (inaccessible via anon key). |
| T-01-05 | Elevation of Privilege | Auth guard loop | mitigate | Guard checks `loading || profileLoading` before any redirect. Race condition between session and profile load prevented by waiting for both flags. See Pitfall 6 in RESEARCH.md. |
| T-01-06 | Elevation of Privilege | profiles.username uniqueness | mitigate | UNIQUE constraint on profiles.username at DB level. Client catches error.code === '23505' and shows "Username taken" — no silent failure. |
| T-01-07 | Denial of Service | friend_invites spam via INSERT | accept | unique(from_user, to_user) constraint prevents duplicate invites from same sender. Rate limiting deferred to Phase 3 (acceptable for Phase 1 with small user base). |
| T-01-08 | Tampering | Storage avatar upload path | mitigate | Upload path set to `{userId}/{timestamp}.ext` where userId = session.user.id from server-issued JWT. Storage RLS update/delete policies check `(storage.foldername(name))[1] = auth.uid()::text` to prevent overwriting another user's avatar. |
| T-01-09 | Tampering | handle_new_user trigger | mitigate | Trigger uses `security definer` + `set search_path = public` to prevent search path injection. `on conflict (id) do nothing` prevents duplicate profile creation. |
