<!-- Pattern 1: CameraView recording (expo-camera 17.x) -->
```typescript
import { CameraView, useCameraPermissions, useMicrophonePermissions } from 'expo-camera';

```

<!-- Pattern 3: Compression (react-native-compressor) -->
```typescript
import { Video as VideoCompressor } from 'react-native-compressor';
const compressed = await VideoCompressor.compress(uri, {
  compressionMethod: 'auto',
  maxSize: 1280,
});
```

<!-- Pattern 4: Upload (ArrayBuffer via expo-file-system + base64-arraybuffer) -->
```typescript
import * as FileSystem from 'expo-file-system';
import { decode } from 'base64-arraybuffer';

const base64 = await FileSystem.readAsStringAsync(fileUri, {
  encoding: FileSystem.EncodingType.Base64,
});
const arrayBuffer = decode(base64);

// XHR with progress:
const xhr = new XMLHttpRequest();
xhr.open('PUT', signedUrl);
xhr.setRequestHeader('Content-Type', 'video/mp4');
xhr.setRequestHeader('x-upsert', 'false');
xhr.upload.onprogress = (e) => { if (e.lengthComputable) onProgress(Math.round(e.loaded/e.total*100)); };
xhr.onload = () => xhr.status >= 200 && xhr.status < 300 ? resolve() : reject(new Error(`${xhr.status}`));
xhr.send(arrayBuffer);
```

<!-- Storage path convention (for RLS compliance) -->
// videos/{friendship_id}/{user_id}/{question_id}.mp4
```

<!-- Anti-patterns to NEVER use -->
// DO NOT use Blob or FormData — produces 0-byte uploads silently on React Native
// DO NOT pass maxDuration: 30000 — units are seconds, not milliseconds
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → Supabase Storage | Signed upload URL; client uploads MP4 directly |
| client → question_responses INSERT | User submits their own response row |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-02-01 | Tampering | question_responses INSERT | mitigate | RLS WITH CHECK (auth.uid() = user_id) from 02-01 migration prevents submitting on behalf of others |
| T-02-02-02 | Tampering | Storage upload path | mitigate | Path includes user_id; Storage INSERT RLS verifies auth.uid() is not null; DELETE RLS verifies segment [2] = user_id |
| T-02-02-03 | Denial of Service | Unbounded video upload size | accept | maxDuration: 30s caps at ~15MB compressed; no additional server-side size limit enforced in MVP |
| T-02-02-04 | Information Disclosure | Signed upload URL | accept | URL is single-use (x-upsert: false) and expires; low risk for 30s window |
| T-02-02-05 | Spoofing | video_url stored as path not signed URL | mitigate | record.tsx stores storage path, not a signed URL; signed URL generated fresh on reveal (RESEARCH.md Pitfall 5) |
</threat_model>
